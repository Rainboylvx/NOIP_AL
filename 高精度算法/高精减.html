高精度减法

朴素高精度减法

在你看完并学会了上面的高精度加法算法后,其时减法和加法差不多,我们只要注意下面几点:

  • s1 的长度 >= s2
  • 正负号问题
  • 最后一位的处理
/*============================================================================
* Title : 朴素高精度减法
* Author: Rainboy
* Time  : 2016-04-14 11:06
* update: 2016-05-09 10:33
* © Copyright 2016 Rainboy. All Rights Reserved.
*=============================================================================*/

/*  原理模拟减法
 *      
 *      过程:
 *        1. 读取字符串
 *        2. 保证s1>=s2 ,a=s1,b=s2
 *        3. 运算 c=a-b
 * */
#include <cstdio>
#include <cstring>

char s1[250],s2[250]; // 存字符串
int a[250],b[250],c[250]; //数据存储

int jianfa(int x[],int y[],int len){
    int i;
    for(i=0;i<len;i++){
        if(x[i] < y[i]) {//借位
            x[i+1]--;
            c[i]=10+x[i]-y[i];
        }
        else{
            c[i]=x[i]-y[i];
        }
    }

    //处理最高位
    int t = len-1;
    while(c[t] == 0 && t >= 0) t--;

    return t+1; //范围最后的位数
}

int main(){
    int la,lb,lc;
    int i;
    int *pa,*pb;
    /* 读取数据 */
    scanf("%s%s",s1,s2);

    /* 处理数据 */
    la = strlen(s1);
    lb = strlen(s2);

    for(i=0;i<la;i++)
        a[i] = s1[la-1-i] - '0';

    for(i=0;i<lb;i++)
        b[i] = s2[lb-1-i] -'0';

    /* 默认 a - b 我们一定保障 pa>=pb*/
    if(lb >la || (la== lb && strcmp(s1,s2)==-1)){ //满足这些条件 a < b
        printf("-");//flag;
        pa = b;pb = a;
        lc =lb;
    } 
    else {
        pa =a;pb=b;
        lc =la;
    }
    int len = jianfa(pa,pb,lc);

    /* 输出 */
    if(len ==0)
        printf("0");
    else {
        for(i=len-1;i>=0;i--)
            printf("%d",c[i]);
    }

    return 0;
}

优化:去掉C数组

实现原理:解决借位的问题

亿进制高精度减法

/*============================================================================

/*  原理模拟减法
 *      
 *      过程:
 *        1. 读取字符串
 *        2. 保证s1>=s2 ,a=s1,b=s2
 *        3. 运算 c=a-b
 * */

#include <cstdio>
#include <cstring>

const int base =8;
const int base2 = 100000000;

char s1[250],s2[250]; // 存字符串

int a[250],b[250],c[250]; //数据存储

int jianfa(int x[],int y[],int len){
    int i,j;
    for(i=0;i<len;i++){
        if(x[i] < y[i]) {//借位
            x[i+1]--;
            c[i]=base2+x[i]-y[i];
        }
        else{
            c[i]=x[i]-y[i];
        }
    }

    //处理最高位
    int t = len-1;
    while(c[t] == 0 && t >= 0) t--;

    return t+1; //范围最后的位数
}

/* 返加 位数 */
int str2num(char x[],int y[]){
    int i,j,k;
    int lenx= strlen(x);

    int xn = lenx / base; //位数
    int xp = lenx % base; //多余的最高位 
    int m = lenx-1;//m 计算用
    for(i=0;i<xn;i++){
        k=1;
        for(j=0;j<8;j++){   //从后往前算
            y[i]+=(x[m-j]-'0')*k;k*=10;
        }
        m-=base;
    }

    if(xp){ //有多余的最高位
        xn++;
        k=1;
        for(j=0;j<xp;j++){
            y[i]+=(x[m-j]-'0')*k;k*=10;
        }
    }

    return xn;
}
int main(){
    int la,lb,lc;
    int i,j;
    int *pa,*pb;
    /* 读取数据 */
    scanf("%s%s",s1,s2);

    /* 处理数据 */
    int lena = str2num(s1,a);
    int lenb = str2num(s2,b);

    la = strlen(s1);
    lb = strlen(s2);
    /* 默认 a - b 我们一定保障 a>b*/
    if(lb >la || (la== lb && strcmp(s1,s2)==-1)){ //满足这些条件 a < b
        printf("-");//flag;
        pa = b;pb = a;
        lc =lb;
    } 
    else {
        pa =a;pb=b;
        lc =la;
    }

    int len = jianfa(pa,pb,lc);

    /* 输出 */
    if(len == 0)
        printf("0");
    else {
        for(i=0;i<len;i++)
            printf("%d",c[i]);
    }


    return 0;
}
